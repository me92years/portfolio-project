pipeline {
    agent any
    environment {
    	FRONTEND_DIR = "${env.WORKSPACE}/frontend"
    }
    stages {
        stage('Build') {
            steps {
                sh """
                	cd ${FRONTEND_DIR}
                	yarn install
                	yarn build
                """
             
            }
        }
        stage('Test') {
            steps {
                echo 'testing the application...'
            }
        }
        stage('Deploy') {
            steps {
                echo 'deploying the application...'
            }
        }
    }
}
